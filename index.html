def calculate(event): a = int(document.getElementById('num1').value) b = int(document.getElementById('num2').value) operator = document.getElementById('operator').value if operator == '+': result = a + b elif operator == '-': result = a - b elif operator == '*': result = a * b elif operator == '/': if b == 0: result = "Error: Division by zero" else: result = a / b else: result = "Invalid operator" document.getElementById('result').innerText = f'Your answer is: {result}' document.getElementById('calculate-btn').addEventListener('click', calculate)
